<script>

  // Simple splash screen logic

  const bootLines = ['Initializing DannyGC Cloud...', 'Connecting to A6000...', 'Mounting UI...'];

  async function runBootSequence() {

    const body = document.getElementById('termBody');

    const cursor = document.getElementById('termCursor');

    for (let i = 0; i < bootLines.length; i++) {

      await new Promise(r => setTimeout(r, 600));

      const line = document.createElement('div'); line.className = 'term-line visible';

      line.innerHTML = `<span class="term-ok">[ OK ]</span><span class="term-text">${bootLines[i]}</span>`;

      body.insertBefore(line, cursor);

    }

    await new Promise(r => setTimeout(r, 800));

    document.getElementById('splash').classList.add('hidden');

    document.getElementById('app').classList.add('visible');

    checkOllama();

  }


  function showTab(name, btn) {

    document.querySelectorAll('.tab-panel').forEach(p => p.classList.remove('active'));

    document.querySelectorAll('.nav-item').forEach(b => b.classList.remove('active'));

    document.getElementById('tab-' + name).classList.add('active');

    if (btn) btn.classList.add('active');

  }


  /* ─── LIVE AI STATUS ─── */

  async function checkOllama() {

    const sidebarDot = document.getElementById('sidebarAiDot');

    const sidebarSt = document.getElementById('ollamaStatus');

    const chatOnline = document.getElementById('chatOnline');

    try {

      const r = await fetch('http://38.247.186.142:11434/api/tags', { signal: AbortSignal.timeout(5000) });

      if (r.ok) {

        sidebarDot.className = 'status-dot green';

        sidebarSt.textContent = 'online'; sidebarSt.style.color = 'var(--green)';

        chatOnline.innerHTML = '<div class="status-dot green"></div> Online';

      } else throw new Error();

    } catch {

      sidebarDot.className = 'status-dot red';

      sidebarSt.textContent = 'offline'; sidebarSt.style.color = 'var(--red)';

      chatOnline.innerHTML = '<div class="status-dot red"></div> Offline (CORS or Server Down)';

    }

  }


  /* ─── AI CHAT ─── */

  async function sendChat() {

    const input = document.getElementById('chatInput');

    const msg = input.value.trim();

    if (!msg) return;

    input.value = '';

    const msgs = document.getElementById('chatMessages');

    msgs.innerHTML += `<div class="chat-msg user"><div class="chat-msg-icon">👤</div><div class="chat-bubble">${msg}</div></div>`;

    

    const bubbleId = 'ai-msg-' + Date.now();

    const thinking = document.createElement('div');

    thinking.className = 'chat-msg';

    thinking.innerHTML = `<div class="chat-msg-icon">☁️</div><div class="chat-bubble" id="${bubbleId}">Processing directive...</div>`;

    msgs.appendChild(thinking);

    msgs.scrollTop = msgs.scrollHeight;

    

    try {

      const res = await fetch('http://38.247.186.142:11434/api/chat', {

        method: 'POST',

        headers: { 'Content-Type': 'application/json' },

        body: JSON.stringify({

          model: 'deepseek-r1:32b',

          messages: [

            { role: 'system', content: 'You are the central intelligence for DannyGC Cloud Services. Your core directive is to execute automated B2B lead enrichment. Keep responses tactical and efficient.' },

            { role: 'user', content: msg }

          ],

          stream: false

        })

      });

      const data = await res.json();

      document.getElementById(bubbleId).innerHTML = data.message.content.replace(/\n/g, '<br>');

    } catch (err) {

      document.getElementById(bubbleId).innerHTML = '<span style="color:var(--red)">System Error: Could not reach DeepSeek model on the A6000.</span>';

    }

    msgs.scrollTop = msgs.scrollHeight;

  }

  

  runBootSequence();
